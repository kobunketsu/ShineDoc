143static inline void* cvAlignPtr( const void* ptr, int align = 32 )
144{
145 CV_DbgAssert ( (align & (align-1)) == 0 );

150{
151 CV_DbgAssert( (align & (align-1)) == 0 && size < INT_MAX );
